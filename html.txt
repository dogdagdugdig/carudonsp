    function formatTime(timeValue) {
  if (!timeValue) return '-';
  
  let hours, minutes;
  
  // ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏ß‡πà‡∏≤‡πÄ‡∏õ‡πá‡∏ô ISO String (‡∏°‡∏µ 'T')
  if (typeof timeValue === 'string' && timeValue.includes('T')) {
    try {
      const date = new Date(timeValue);
      // ‡πÉ‡∏ä‡πâ‡πÄ‡∏ß‡∏•‡∏≤‡∏ó‡πâ‡∏≠‡∏á‡∏ñ‡∏¥‡πà‡∏ô (Local Time) ‡∏Ç‡∏≠‡∏á‡πÄ‡∏ö‡∏£‡∏≤‡∏ß‡πå‡πÄ‡∏ã‡∏≠‡∏£‡πå (‡∏õ‡∏£‡∏∞‡πÄ‡∏ó‡∏®‡πÑ‡∏ó‡∏¢)
      hours = date.getHours().toString().padStart(2, '0');
      minutes = date.getMinutes().toString().padStart(2, '0');
    } catch (e) {
      return timeValue;
    }
  } 
  // ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏ß‡πà‡∏≤‡πÄ‡∏õ‡πá‡∏ô‡∏£‡∏π‡∏õ‡πÅ‡∏ö‡∏ö "HH:mm" ‡∏à‡∏≤‡∏Å input type="time"
  else if (typeof timeValue === 'string' && timeValue.includes(':')) {
    const parts = timeValue.split(':');
    hours = parts[0].padStart(2, '0');
    minutes = parts[1].padStart(2, '0');
  } 
  else {
    return timeValue;
  }
  
  // ‡∏Ñ‡∏∑‡∏ô‡∏Ñ‡πà‡∏≤‡∏£‡∏π‡∏õ‡πÅ‡∏ö‡∏ö "14.30 ‡∏ô."
  return `${hours}.${minutes} ‡∏ô.`;
}

    function updateMonthlyStats(bookings) {
      const table = document.getElementById('monthlyStatsTable');
      const now = new Date();
      const currentMonth = now.getMonth();
      const currentYear = now.getFullYear();

      const monthlyBookings = bookings.filter(b => {
        const bookingDate = new Date(b.startDate);
        return bookingDate.getMonth() === currentMonth && bookingDate.getFullYear() === currentYear;
      });

      const carStats = {};
      CARS.forEach(car => {
        carStats[car.name] = { count: 0, mileage: 0 };
      });

      monthlyBookings.forEach(b => {
        if (carStats[b.car]) {
          carStats[b.car].count++;
          carStats[b.car].mileage += parseInt(b.mileage) || 0;
        }
      });

      table.innerHTML = Object.entries(carStats).map(([car, stats]) => `
        <tr class="border-b border-gray-100 hover:bg-orange-50/50 transition-colors">
          <td class="p-3 font-kanit">${car}</td>
          <td class="p-3 text-center font-kanit font-semibold">${stats.count}</td>
          <td class="p-3 text-center font-kanit">${stats.mileage.toLocaleString()} ‡∏Å‡∏°.</td>
        </tr>
      `).join('');
    }

    function renderStatsChart() {
      const ctx = document.getElementById('statsChart').getContext('2d');
      
      if (statsChart) {
        statsChart.destroy();
      }

      const last7Days = [];
      const today = new Date();
      for (let i = 6; i >= 0; i--) {
        const date = new Date(today);
        date.setDate(date.getDate() - i);
        last7Days.push(date.toISOString().split('T')[0]);
      }

      const datasets = CARS.map(car => {
        const data = last7Days.map(date => {
          return allBookings.filter(b => b.startDate === date && b.car === car.name).length;
        });
        return {
          label: car.name,
          data: data,
          backgroundColor: car.color,
          borderColor: car.color,
          borderWidth: 1,
          borderRadius: 6
        };
      });

      const labels = last7Days.map(date => {
        const d = new Date(date);
        return d.toLocaleDateString('th-TH', { day: 'numeric', month: 'short' });
      });

      statsChart = new Chart(ctx, {
        type: 'bar',
        data: { labels, datasets },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: {
            legend: {
              position: 'bottom',
              labels: {
                font: { family: 'Kanit', size: 11 },
